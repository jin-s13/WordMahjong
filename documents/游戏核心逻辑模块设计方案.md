# 游戏核心逻辑模块设计方案

## 1. 游戏流程控制

### 1.1 流程概述
游戏流程控制负责管理游戏的整个生命周期，包括游戏初始化、回合循环、操作响应和游戏结束等环节。流程控制遵循以下原则：
- 严格按照麻将传统规则进行回合交替
- 确保每个玩家的操作顺序正确
- 处理公共响应阶段（胡牌>吃牌优先级）
- 管理牌墙和出牌历史

### 1.2 核心流程实现

```typescript
// 类型定义
interface Card {
  id: string;
  char: string;
  category: string;
  frequency: number;
}

interface Player {
  id: number;
  name: string;
  hand: Card[];
  eatenCards: Card[][];
  isWin: boolean;
}

interface GameState {
  players: Player[];
  wall: Card[];
  discardPile: Card[];
  currentDiscard: Card | null;
  history: Action[];
  timer: NodeJS.Timeout | null;
  winner: number | null;
}

interface Action {
  type: 'draw' | 'discard' | 'eat' | 'win';
  playerId: number;
  card?: Card;
  details?: any;
  timestamp: Date;
}

// 游戏状态管理
class GameFlowController {
  private playerCount: number;
  private currentPlayerIndex: number;
  private gameStatus: 'waiting' | 'playing' | 'finished';
  private roundCount: number;
  private gameState: GameState;

  constructor(playerCount: number) {
    this.playerCount = playerCount;
    this.currentPlayerIndex = 0; // 庄家为0号玩家
    this.gameStatus = 'waiting'; // 状态：waiting/playing/finished
    this.roundCount = 1;
  }

  // 初始化游戏
  initializeGame(): void {
    // 1. 初始化牌库
    const deck: Card[] = generateDeck();
    // 2. 洗牌
    const shuffledDeck: Card[] = shuffle(deck);
    // 3. 发牌
    const { players, wall } = dealCards(shuffledDeck, this.playerCount);
    // 4. 设置游戏状态
    this.gameState = {
      players,
      wall,
      discardPile: [],
      currentDiscard: null,
      history: [],
      timer: null,
      winner: null
    };
    this.gameStatus = 'playing';
  }

  // 开始新回合
  startNewTurn(): void {
    const currentPlayer = this.gameState.players[this.currentPlayerIndex];
    
    // 摸牌
    const drawnCard = this.gameState.wall.pop();
    if (!drawnCard) {
      this.endGame('draw'); // 流局
      return;
    }
    currentPlayer.hand.push(drawnCard);
    this.recordAction('draw', this.currentPlayerIndex, drawnCard);

    // 检查是否可以胡牌
    if (this.checkWinCondition(currentPlayer)) {
      this.endGame('win', this.currentPlayerIndex);
      return;
    }
  }

  // 处理出牌
  handleDiscard(playerId: number, cardId: string): void {
    const player = this.gameState.players[playerId];
    const cardIndex = player.hand.findIndex(card => card.id === cardId);
    
    if (cardIndex === -1) {
      throw new Error('无效的出牌：玩家手牌中不存在该牌');
    }

    const discardedCard = player.hand.splice(cardIndex, 1)[0];
    this.gameState.discardPile.push(discardedCard);
    this.gameState.currentDiscard = discardedCard;
    this.recordAction('discard', playerId, discardedCard);

    // 公共响应阶段
    this.publicResponsePhase();
  }

  // 公共响应阶段
  publicResponsePhase(): void {
    // 1. 检查是否有玩家胡牌
    for (let i = 0; i < this.playerCount; i++) {
      if (i === this.currentPlayerIndex) continue; // 跳过当前出牌玩家
      const player = this.gameState.players[i];
      if (this.checkWinCondition(player)) {
        this.endGame('win', i);
        return;
      }
    }

    // 2. 检查是否有玩家可以吃牌（仅下家）
    const nextPlayerIndex = (this.currentPlayerIndex + 1) % this.playerCount;
    const nextPlayer = this.gameState.players[nextPlayerIndex];
    if (this.checkEatCondition(nextPlayer)) {
      // 显示吃牌选项
      this.showEatOption(nextPlayerIndex);
      return;
    }

    // 3. 进入下一回合
    this.nextTurn();
  }

  // 下一回合
  nextTurn(): void {
    this.currentPlayerIndex = (this.currentPlayerIndex + 1) % this.playerCount;
    this.startNewTurn();
  }

  // 结束游戏
  endGame(reason: 'win' | 'draw', winnerIndex: number | null = null): void {
    this.gameStatus = 'finished';
    if (reason === 'win' && winnerIndex !== null) {
      this.gameState.players[winnerIndex].isWin = true;
      this.gameState.winner = winnerIndex;
    }
    // 清理定时器
    if (this.gameState.timer) {
      clearTimeout(this.gameState.timer);
    }
  }

  // 记录操作
  recordAction(type: Action['type'], playerId: number, card?: Card, details: any = {}): void {
    this.gameState.history.push({
      type,
      playerId,
      card,
      details,
      timestamp: new Date()
    });
  }
```

## 2. 牌库管理

### 2.1 牌库结构
牌库管理负责维护游戏中的牌墙、手牌、吃牌和出牌历史。牌库结构包括：
- **牌墙**：剩余未摸的牌
- **手牌**：每个玩家当前持有的牌
- **吃牌**：玩家明放的吃牌组合
- **出牌历史**：所有玩家的出牌记录

### 2.2 核心功能实现

```typescript
// 牌库管理类
class CardManager {
  private wall: Card[];
  private discardPile: Card[];

  constructor(deck: Card[]) {
    this.wall = [...deck];
    this.discardPile = [];
  }

  // 摸牌
  drawCard(): Card | null {
    if (this.wall.length === 0) {
      return null; // 牌墙已空
    }
    return this.wall.pop()!;
  }

  // 出牌
  discardCard(card: Card): number {
    this.discardPile.push(card);
    return this.discardPile.length - 1;
  }

  // 获取牌墙信息
  getWallInfo(): { remainingCount: number; totalCount: number } {
    return {
      remainingCount: this.wall.length,
      totalCount: 144 - this.discardPile.length - this.wall.length
    };
  }

  // 获取出牌历史
  getDiscardHistory(): Card[] {
    return [...this.discardPile];
  }

  // 验证牌的合法性
  validateCard(card: Card): boolean {
    // 检查牌是否符合游戏规则
    const validCategories = [
      '情感核心字', '高频动词', '日常名词', '形容词',
      '副词虚词语气词', '场景方位字', '补充融合字'
    ];
    return validCategories.includes(card.category) && card.char.length === 1;
  }

  // 牌型分析（用于AI决策）
  analyzeHandCards(cards: Card[]): { categoryCount: any; posAnalysis: any; combinationScore: number } {
    // 分类统计
    const categoryCount: any = {};
    cards.forEach(card => {
      categoryCount[card.category] = (categoryCount[card.category] || 0) + 1;
    });

    // 词性分析
    const posAnalysis = this.analyzePartOfSpeech(cards);

    // 组合潜力评估
    const combinationScore = this.evaluateCombinationPotential(cards);

    return {
      categoryCount,
      posAnalysis,
      combinationScore
    };
  }

  // 词性分析
  analyzePartOfSpeech(cards: Card[]): any {
    const posMap: any = {
      '情感核心字': 'pronouns',
      '高频动词': 'verbs',
      '日常名词': 'nouns',
      '形容词': 'adjectives',
      '副词虚词语气词': 'particles',
      '场景方位字': 'nouns',
      '补充融合字': 'numerals'
    };

    const posAnalysis: any = {
      nouns: [],
      verbs: [],
      adjectives: [],
      adverbs: [],
      pronouns: [],
      particles: [],
      numerals: [],
      other: []
    };

    cards.forEach(card => {
      const pos = posMap[card.category] || 'other';
      posAnalysis[pos].push(card);
    });

    return posAnalysis;
  }

  // 组合潜力评估
  evaluateCombinationPotential(cards: Card[]): number {
    const highFrequencyWords = cards.filter(card => card.frequency >= 4);
    const coreWords = cards.filter(card => ['我', '你', '爱', '家', '心', '吃', '说', '想'].includes(card.char));
    const functionWords = cards.filter(card => ['的', '了', '就', '也'].includes(card.char));

    let score = 0;
    score += Math.min(30, (highFrequencyWords.length / cards.length) * 30);
    score += Math.min(30, coreWords.length * 5);
    score += Math.min(25, functionWords.length * 5);
    
    const uniqueCategories = new Set(cards.map(card => card.category));
    score += Math.min(15, uniqueCategories.size * 2);

    return Math.min(100, score);
  }
}
```

## 3. 玩家行为处理

### 3.1 玩家行为类型
玩家行为处理负责处理玩家的各种操作，包括：
- 摸牌
- 出牌
- 吃牌
- 胡牌
- 选择/取消选择牌

### 3.2 核心功能实现

```typescript
// 玩家行为管理器
class PlayerActionHandler {
  private gameState: GameState;

  constructor(gameState: GameState) {
    this.gameState = gameState;
  }

  // 摸牌
  handleDraw(playerId: number): Card {
    const player = this.gameState.players[playerId];
    const drawnCard = this.gameState.wall.pop();
    
    if (!drawnCard) {
      throw new Error('牌墙已空，无法摸牌');
    }

    player.hand.push(drawnCard);
    return drawnCard;
  }

  // 出牌
  handleDiscard(playerId: number, cardId: string): Card {
    const player = this.gameState.players[playerId];
    const cardIndex = player.hand.findIndex(card => card.id === cardId);
    
    if (cardIndex === -1) {
      throw new Error('无效的出牌：玩家手牌中不存在该牌');
    }

    const discardedCard = player.hand.splice(cardIndex, 1)[0];
    this.gameState.discardPile.push(discardedCard);
    this.gameState.currentDiscard = discardedCard;
    return discardedCard;
  }

  // 吃牌
  handleEat(playerId: number, cardId: string, combination: Card[]): Card[] {
    const player = this.gameState.players[playerId];
    const cardIndex = player.hand.findIndex(card => card.id === cardId);
    
    if (cardIndex === -1) {
      throw new Error('无效的吃牌：玩家手牌中不存在该牌');
    }

    const eatenCard = player.hand.splice(cardIndex, 1)[0];
    player.eatenCards.push([...combination, eatenCard]);
    return [...combination, eatenCard];
  }

  // 胡牌
  handleWin(playerId: number): void {
    const player = this.gameState.players[playerId];
    player.isWin = true;
    this.gameState.winner = playerId;
  }

  // 选择/取消选择牌
  toggleCardSelection(cardId: string, playerId: number, isEaten: boolean = false): Card {
    const player = this.gameState.players[playerId];
    const cards = isEaten ? player.eatenCards.flat() : player.hand;
    const card = cards.find(c => c.id === cardId);

    if (!card) {
      throw new Error('无效的牌选择');
    }

    return card;
  }

  // 验证操作合法性
  validateAction(actionType: string, playerId: number, params: any = {}): boolean {
    const currentPlayer = this.gameState.players[this.gameState.currentPlayerIndex];
    
    // 检查操作权限
    if (actionType !== 'eat' && currentPlayer.id !== playerId) {
      return false; // 不是当前玩家的回合
    }

    // 检查吃牌权限（仅下家可吃）
    if (actionType === 'eat') {
      const previousPlayerIndex = (this.gameState.currentPlayerIndex - 1 + this.gameState.players.length) % this.gameState.players.length;
      const previousPlayer = this.gameState.players[previousPlayerIndex];
      
      if (!this.gameState.currentDiscard) {
        return false; // 没有可吃的牌
      }

      // 检查玩家是否为下家
      const nextPlayerIndex = (this.gameState.currentPlayerIndex + 1) % this.gameState.players.length;
      if (nextPlayerIndex !== playerId) {
        return false; // 不是下家，不能吃牌
      }
    }

    return true;
  }
}
```

## 4. AI对手行为处理

### 4.1 AI行为概述
AI对手行为处理负责模拟人类玩家的行为，包括：
- 手牌分析和评估
- 决策制定（吃牌/胡牌/出牌）
- 行动执行

### 4.2 核心功能实现

```typescript
// AI行为处理器
class AIBehaviorHandler {
  private aiLevel: string;
  private decisionDelay: number;

  constructor(aiLevel: string = 'normal') {
    this.aiLevel = aiLevel;
    this.decisionDelay = this.getDecisionDelay();
  }

  // 获取决策延迟时间
  getDecisionDelay(): number {
    const delays: any = {
      'easy': 500, // 简单模式：快速决策
      'normal': 1500, // 正常模式：中等延迟
      'hard': 2500 // 困难模式：长时间思考
    };
    return delays[this.aiLevel] || 1500;
  }

  // AI决策过程
  async makeDecision(gameState: GameState, playerId: number): Promise<any> {
    await this.delay(this.decisionDelay);
    
    const player = gameState.players[playerId];
    
    // 1. 检查是否可以胡牌
    if (this.checkWinCondition(player)) {
      return { type: 'win' };
    }

    // 2. 检查是否可以吃牌
    if (this.checkEatCondition(player, gameState)) {
      return { type: 'eat', details: this.findBestEatCombination(player, gameState) };
    }

    // 3. 决定出牌
    return { type: 'discard', details: this.findBestDiscardCard(player) };
  }

  // 检查胡牌条件
  checkWinCondition(player: Player): boolean {
    if (player.hand.length + player.eatenCards.flat().length !== 14) {
      return false; // 牌数不足
    }

    const allCards = [...player.hand, ...player.eatenCards.flat()];
    const sentence = allCards.map(card => card.char).join('');
    
    return this.validateSentence(sentence);
  }

  // 检查吃牌条件
  checkEatCondition(player: Player, gameState: GameState): boolean {
    if (!gameState.currentDiscard) {
      return false; // 没有可吃的牌
    }

    // 简单的语义匹配检查
    const possibleCombinations = this.findPossibleEatCombinations(player, gameState);
    return possibleCombinations.length > 0;
  }

  // 寻找最佳吃牌组合
  findBestEatCombination(player: Player, gameState: GameState): Card[] | null {
    const possibleCombinations = this.findPossibleEatCombinations(player, gameState);
    if (possibleCombinations.length === 0) return null;

    // 选择得分最高的组合
    const scoredCombinations = possibleCombinations.map(combination => {
      const score = this.evaluateEatCombination(combination);
      return { combination, score };
    });

    return scoredCombinations.sort((a, b) => b.score - a.score)[0].combination;
  }

  // 寻找最佳出牌
  findBestDiscardCard(player: Player): Card | null {
    let bestCard: Card | null = null;
    let lowestScore = Infinity;

    player.hand.forEach(card => {
      // 模拟移除该牌后的手牌评估
      const tempHand = player.hand.filter(c => c.id !== card.id);
      const score = this.evaluateHand(tempHand);

      if (score < lowestScore) {
        lowestScore = score;
        bestCard = card;
      }
    });

    return bestCard;
  }

  // 评估吃牌组合
  evaluateEatCombination(combination: Card[]): number {
    // 简单的语义和组合评估
    const chars = combination.map(card => card.char);
    const combinationStr = chars.join('');

    // 常用搭配权重
    const commonCollocations = ['吃饭', '喝水', '看书', '听音乐', '说话', '写字'];
    const collocationScore = commonCollocations.reduce((score, collocation) => {
      if (combinationStr.includes(collocation)) {
        return score + 10;
      }
      return score;
    }, 0);

    // 词性多样性权重
    const uniqueCategories = new Set(combination.map(card => card.category));
    const categoryScore = uniqueCategories.size * 5;

    return collocationScore + categoryScore;
  }

  // 评估手牌
  evaluateHand(cards: Card[]): number {
    // 手牌价值评估
    const cardValueMap = this.getCardValueMap();
    let totalValue = 0;

    cards.forEach(card => {
      totalValue += cardValueMap[card.char] || 1;
    });

    return totalValue;
  }

  // 获取牌价值映射
  getCardValueMap(): any {
    // 核心词价值最高
    return {
      '我': 10, '你': 10, '爱': 10, '家': 10, '心': 10,
      '吃': 9, '说': 9, '想': 9,
      '的': 8, '了': 8, '就': 8, '也': 8,
      '好': 7, '美': 7, '香': 7, '甜': 7,
      // 其他字价值较低
    };
  }

  // 延迟函数
  delay(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }

  // 寻找可能的吃牌组合
  private findPossibleEatCombinations(player: Player, gameState: GameState): Card[][] {
    const possibleCombinations: Card[][] = [];
    
    if (!gameState.currentDiscard) {
      return possibleCombinations;
    }
    
    const discardedCard = gameState.currentDiscard;
    
    // 查找包含废弃牌的2-3字组合
    player.hand.forEach(card => {
      // 检查是否可以与废弃牌组成词语
      const combination1 = [card, discardedCard];
      if (this.isPossibleWordCombination(combination1)) {
        possibleCombinations.push(combination1);
      }

      // 检查是否可以与废弃牌和另一张牌组成词语
      player.hand.forEach(otherCard => {
        if (otherCard.id === card.id) return;
        
        const combination2 = [card, otherCard, discardedCard];
        if (this.isPossibleWordCombination(combination2)) {
          possibleCombinations.push(combination2);
        }
      });
    });

    return possibleCombinations;
  }

  // 检查是否为可能的词语组合
  private isPossibleWordCombination(combination: Card[]): boolean {
    const chars = combination.map(card => card.char).join('');
    
    // 常用词语匹配
    const commonWords = [
      '吃饭', '喝水', '看书', '听音乐', '说话', '写字', '做饭', '开门',
      '关门', '拿钱', '放东西', '想家', '想念', '买东西', '卖东西',
      '学习', '睡觉', '飞行', '洗澡', '爱心', '家人', '心理', '日常'
    ];

    return commonWords.some(word => {
      // 检查组合是否包含该词语
      const wordChars = word.split('');
      return wordChars.every(char => chars.includes(char));
    });
  }

  // 验证句子语义正确性
  private validateSentence(sentence: string): boolean {
    // 简单的句子验证
    const validPatterns = [
      // 主语+谓语+宾语
      /^[我你他她它]们?[吃玩看听读写想爱]了?[饭水书音乐话字饭门东西家学习觉]$/,
      // 带形容词的句子
      /^[我你他她它]们?[吃玩看听读写想爱]了?[好美的香的甜的快的慢的大的小的]?[饭水书音乐话字饭门东西家学习觉]$/,
      // 带时间/地点的句子
      /^[我你他她它]们?[在从去来][家里学校店里公园厨房客厅房间里外上下左右前后东南西北]?[吃玩看听读写想爱]了?[饭水书音乐话字饭门东西家学习觉]$/
    ];

    return validPatterns.some(pattern => pattern.test(sentence));
  }
}
```

## 5. 吃牌和胡牌判定

### 5.1 判定机制
吃牌和胡牌判定是AI裁判的核心功能，负责检查操作的语义合规性。判定机制包括：
- 吃牌判定：检查是否能组成符合现代汉语语义的词语
- 胡牌判定：检查是否能组成完整通顺的句子
- 语义验证：检查词语组合和句子的语义合规性

### 5.2 核心功能实现

```typescript
      // 检查是否可以与废弃牌组成词语
      const combination1 = [card, discardedCard];
      if (this.isPossibleWordCombination(combination1)) {
        possibleCombinations.push(combination1);
      }

      // 检查是否可以与废弃牌和另一张牌组成词语
      player.hand.forEach(otherCard => {
        if (otherCard.id === card.id) return;
        
        const combination2 = [card, otherCard, discardedCard];
        if (this.isPossibleWordCombination(combination2)) {
          possibleCombinations.push(combination2);
        }
      });
    });

    return possibleCombinations;
  }

  // 检查是否为可能的词语组合
  isPossibleWordCombination(combination: Card[]): boolean {
    const chars = combination.map(card => card.char).join('');
    
    // 常用词语匹配
    const commonWords = [
      '吃饭', '喝水', '看书', '听音乐', '说话', '写字', '做饭', '开门',
      '关门', '拿钱', '放东西', '想家', '想念', '买东西', '卖东西',
      '学习', '睡觉', '飞行', '洗澡', '爱心', '家人', '心理', '日常'
    ];

    return commonWords.some(word => {
      // 检查组合是否包含该词语
      const wordChars = word.split('');
      return wordChars.every(char => chars.includes(char));
    });
  }

  // 验证组合语义正确性
  isValidCombination(combinationStr: string): boolean {
    // 简单的语义验证
    const validCombinations = [
      '吃饭', '喝水', '看书', '听音乐', '说话', '写字', '做饭', '开门',
      '关门', '拿钱', '放东西', '想家', '想念', '买东西', '卖东西',
      '学习', '睡觉', '飞行', '洗澡', '爱心', '家人', '心理', '日常'
    ];

    return validCombinations.includes(combinationStr);
  }

  // 验证句子语义正确性
  isValidSentence(sentence: string): boolean {
    // 简单的句子验证
    const validPatterns = [
      // 主语+谓语+宾语
      /^[我你他她它]们?[吃玩看听读写想爱]了?[饭水书音乐话字饭门东西家学习觉]$/,
      // 带形容词的句子
      /^[我你他她它]们?[吃玩看听读写想爱]了?[好美的香的甜的快的慢的大的小的]?[饭水书音乐话字饭门东西家学习觉]$/,
      // 带时间/地点的句子
      /^[我你他她它]们?[在从去来][家里学校店里公园厨房客厅房间里外上下左右前后东南西北]?[吃玩看听读写想爱]了?[饭水书音乐话字饭门东西家学习觉]$/
    ];

    return validPatterns.some(pattern => pattern.test(sentence));
  }

  // 自动添加标点符号
  addPunctuation(sentence: string): string {
    // 简单的标点符号添加规则
    if (sentence.endsWith('吗') || sentence.endsWith('呢') || sentence.endsWith('吧')) {
      return sentence + '？';
    } else if (sentence.endsWith('啊') || sentence.endsWith('呀')) {
      return sentence + '！';
    } else {
      return sentence + '。';
    }
  }
}
```

```typescript
    const errorMessages: any = {
      'PERMISSION_DENIED': '操作权限不足',
      'INVALID_CARD': '无效的牌选择',
      'WALL_EMPTY': '牌墙已空，游戏结束',
      'INSUFFICIENT_CARDS': '手牌数不足',
      'SEMANTIC_ERROR': '语义验证失败'
    };

    return errorMessages[error.code] || '未知错误';
  }

  // 边界条件检查
  checkBoundaryConditions(gameState: GameState): void {
    // 检查牌墙是否为空
    if (gameState.wallCards.length === 0) {
      throw this.handleEmptyWallError();
    }

    // 检查玩家手牌数
    gameState.players.forEach(player => {
      if (player.handCards.length + player.eatenCards.flat().length < 13) {
        throw this.handleInsufficientCardsError(player.id);
      }
    });

    // 检查当前回合是否有效
    if (gameState.currentPlayerIndex < 0 || gameState.currentPlayerIndex >= gameState.players.length) {

    }

      return false;
    }

    // 检查牌总数
    const totalCards = gameState.wallCards.length +
      gameState.players.reduce((sum, player) => 
        sum + player.handCards.length + player.eatenCards.flat().length, 0) +
      gameState.discardPile.length;

    return totalCards === 144; // 标准牌数
  }
}
```

### 8.1 洗牌算法

```typescript
// Fisher-Yates洗牌算法
const shuffle = (array: Card[]): Card[] => {
  let currentIndex = array.length, randomIndex;
  
  while (currentIndex != 0) {
    randomIndex = Math.floor(Math.random() * currentIndex);
    currentIndex--;
    [array[currentIndex], array[randomIndex]] = [
      array[randomIndex], array[currentIndex]];
  }

  return array;
};
```

### 8.2 发牌算法
按照传统麻将规则发牌，每人连续摸3轮，每轮4张牌，庄家多摸1张。

```typescript
// 发牌算法
const dealCards = (deck: Card[], playerCount: number = 4): { players: Player[], wall: Card[] } => {
  const players: Player[] = Array.from({ length: playerCount }, (_, i) => ({
    id: i,
    handCards: [],
    eatenCards: [],
    name: `玩家 ${i + 1}`,
    isWin: false
  }));
  
  const wall = [...deck];
  
  // 每人连续摸3轮，每轮4张牌
  for (let round = 0; round < 3; round++) {
    for (let p = 0; p < playerCount; p++) {
      players[p].handCards.push(wall.pop()!, wall.pop()!, wall.pop()!, wall.pop()!);
    }
  }
  
  // 庄家跳牌摸2张，其余玩家各补摸1张
  players[0].handCards.push(wall.pop()!, wall.pop()!);
  for (let p = 1; p < playerCount; p++) {
    players[p].handCards.push(wall.pop()!);
  }
  
  return { players, wall };
};
```

```typescript
// 语义匹配算法
const semanticMatching = (combination: Card[]): Card[][] => {
  // 常用词语匹配库
  const wordLibrary = [
    '吃饭', '喝水', '看书', '听音乐', '说话', '写字', '做饭', '开门',
    '关门', '拿钱', '放东西', '想家', '想念', '买东西', '卖东西',
    '学习', '睡觉', '飞行', '洗澡', '爱心', '家人', '心理', '日常'
  ];

  // 组合字符排序（考虑顺序）
  const sortedCombination = combination.sort((a, b) => {
    // 根据词库中的出现顺序排序
    const indexA = wordLibrary.findIndex(word => word.includes(a.char));
    const indexB = wordLibrary.findIndex(word => word.includes(b.char));
    return indexA - indexB;
  });

  // 生成可能的组合
  const possibleCombinations: Card[][] = [];
  
  // 生成所有可能的连续组合
  for (let i = 0; i < sortedCombination.length; i++) {
    for (let j = i + 1; j <= sortedCombination.length; j++) {
      const subCombination = sortedCombination.slice(i, j);
      const charSequence = subCombination.map(card => card.char).join('');
      
      if (wordLibrary.includes(charSequence)) {
        possibleCombinations.push(subCombination);
      }
    }
  }

  return possibleCombinations;
};
```

```typescript
const analyzeHandPattern = (cards: Card[]): any => {

  const balanceScore = calculateHandBalance(posCount);

// 计算手牌平衡度
const calculateHandBalance = (posCount: any): number => {
  // 理想的词性分布
  const idealDistribution = {
    nouns: 4, verbs: 3, adjectives: 2,
    adverbs: 2, pronouns: 1, particles: 1,
    numerals: 1, other: 0
  };

  let totalDeviation = 0;
  Object.keys(idealDistribution).forEach(pos => {
    const deviation = Math.abs(posCount[pos] - idealDistribution[pos]);
    totalDeviation += deviation;
  });

  // 转换为0-100分，偏差越小分数越高
  return Math.max(0, 100 - (totalDeviation * 10));
};

// 评估组合潜力
const evaluateCombinationPotential = (cards: Card[]): number => {
  const highFrequencyWords = cards.filter(card => card.frequency >= 4);
  const coreWords = cards.filter(card => ['我', '你', '爱', '家', '心', '吃', '说', '想'].includes(card.char));
  const functionWords = cards.filter(card => ['的', '了', '就', '也'].includes(card.char));

  let score = 0;
  score += Math.min(30, (highFrequencyWords.length / cards.length) * 30);
  score += Math.min(30, coreWords.length * 5);
  score += Math.min(25, functionWords.length * 5);
  
  const uniqueCategories = new Set(cards.map(card => card.category));
  score += Math.min(15, uniqueCategories.size * 2);

  return Math.min(100, score);
};
```
